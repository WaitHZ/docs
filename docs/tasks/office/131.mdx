    """Validate if string is a valid IP address"""
    parts = ip_str.split('.')

    try:

        return False

            return '/hidden/'

import re

# Define patterns for sensitive information
patterns = {
    'phone': r'\b(\+?1[-.\s]?)?\(?\d{3}\)?[-.\s]?\d{3}[-.\s]?\d{4}\b',
    'ssn': r'\b\d{3}-\d{2}-\d{4}\b',
    'email': r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b',
    'credit_card': r'\b\d{4}[-\s]?\d{4}[-\s]?\d{4}[-\s]?\d{4}\b|\b\d{15,16}\b',
    'ip_address': r'\b(?:\d{1,3}\.){3}\d{1,3}\b'
}

def desensitize_text(text):
    """Desensitize sensitive information in text"""
    # Phone numbers
    text = re.sub(patterns['phone'], '/hidden/', text)
    
    # SSNs
    text = re.sub(patterns['ssn'], '/hidden/', text)
    
    # Email addresses
    text = re.sub(patterns['email'], '/hidden/', text)
    
    # Credit card numbers
    text = re.sub(patterns['credit_card'], '/hidden/', text)
    
    # IP addresses
    text = re.sub(patterns['ip_address'], '/hidden/', text)
    
    return text

def desensitize_json(data):
    """Recursively desensitize sensitive information in JSON data"""
    if isinstance(data, dict):
        return {key: desensitize_json(value) for key, value in data.items()}
    elif isinstance(data, list):
        return [desensitize_json(item) for item in data]
    elif isinstance(data, str):
        return desensitize_text(data)
    else:
        return data
