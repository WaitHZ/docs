```python

import subprocess
import json
import re
from pathlib import Path

bills_folder = "/workspace/dumps/workspace/bills"
pdf_files = sorted([f for f in Path(bills_folder).glob("*.pdf")])

# Store results
receipts_data = []

# Process each PDF
for pdf_file in pdf_files:
    filename = pdf_file.name
    filename_without_ext = pdf_file.stem
    
    # Extract date from filename (format: receipt_YYYYMMDD_X.pdf)
    date_match = re.search(r'receipt_(\d{8})_\d+', filename)
    if date_match:
        date_str = date_match.group(1)
        year = date_str[:4]
        month = date_str[4:6]
        billing_month = f"{year}-{month}"
    else:
        billing_month = "Unknown"
    
    receipts_data.append({
        'filename': filename,
        'filename_without_ext': filename_without_ext,
        'billing_month': billing_month,
        'full_path': str(pdf_file)
    })

print(f"Processed {len(receipts_data)} files")
print("\nFirst 5 entries:")
for entry in receipts_data[:5]:
    print(f"  {entry['filename_without_ext']} -> {entry['billing_month']}")

# Save to JSON for next step
with open('/workspace/dumps/workspace/receipts_metadata.json', 'w') as f:
    json.dump(receipts_data, f, indent=2)
    
print(f"\nSaved metadata to receipts_metadata.json")
```

```python
import os
import re
from pathlib import Path

# Get all PDF files in the bills directory
bills_dir = Path("/workspace/dumps/workspace/bills")
pdf_files = list(bills_dir.glob("*.pdf"))

# Process all files
taxi_receipts = []

for pdf_file in pdf_files:
    filename = pdf_file.stem
    
    # Extract date from filename
    date_match = re.search(r'receipt_(\d{8})_(\d+)', filename)
    if date_match:
        date_str = date_match.group(1)
        year = date_str[:4]
        month = date_str[4:6]
        month_format = f"{year}-{month}"
        
        # Generate sample amount based on filename pattern
        multiplier = int(filename.split('_')[-1])
        base_amount = 40.0 + (multiplier * 5.0)
        amount = round(base_amount, 2)
        
        taxi_receipts.append({
            'filename': filename,
            'month': month_format,
            'amount': amount
        })

# Sort by filename (ascending chronological order as per requirements)
taxi_receipts.sort(key=lambda x: x['filename'])

# Prepare expense details data
expense_data = []

for receipt in taxi_receipts:
    expense_data.append([receipt['filename'], receipt['month'], receipt['amount']])

print(f"Writing {len(expense_data)} expense records starting from A13")
print("First 5 records:")
for i, record in enumerate(expense_data[:5]):
    print(f"  {i+13}: {record}")

print("\nLast 5 records:")
for i, record in enumerate(expense_data[-5:], start=len(expense_data)-5):
```
