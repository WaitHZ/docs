bills_folder = "/workspace/dumps/workspace/bills"
pdf_files = sorted([f for f in Path(bills_folder).glob("*.pdf")])

# Store results
receipts_data = []

# Process each PDF
for pdf_file in pdf_files:
    filename = pdf_file.name
    filename_without_ext = pdf_file.stem
    
    # Extract date from filename (format: receipt_YYYYMMDD_X.pdf)
    date_match = re.search(r'receipt_(\d{8})_\d+', filename)
    if date_match:
        date_str = date_match.group(1)
        year = date_str[:4]
        month = date_str[4:6]
        billing_month = f"{year}-{month}"
    else:
        billing_month = "Unknown"
    
    receipts_data.append({
        'filename': filename,
        'filename_without_ext': filename_without_ext,
        'billing_month': billing_month,
        'full_path': str(pdf_file)
    })

print(f"Processed {len(receipts_data)} files")
print("\nFirst 5 entries:")
for entry in receipts_data[:5]:
    print(f"  {entry['filename_without_ext']} -> {entry['billing_month']}")

# Save to JSON for next step
with open('/workspace/dumps/workspace/receipts_metadata.json', 'w') as f:
    json.dump(receipts_data, f, indent=2)
    
print(f"\nSaved metadata to receipts_metadata.json")
