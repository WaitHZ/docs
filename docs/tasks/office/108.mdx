# Transform the data from wide format to long format
import pandas as pd
import numpy as np

# Read the data
df = pd.read_excel("/workspace/dumps/workspace/Household_Appliances.xlsx")

# Skip the header row and get actual data
data_rows = df.iloc[1:].copy()

# Reset index
data_rows = data_rows.reset_index(drop=True)

# Create the transformed dataset
transformed_data = []

# Process each row
for idx, row in data_rows.iterrows():
    time_value = row.iloc[0]  # Time column
    
    # Skip rows with NaT or NaN time values
    if pd.isna(time_value):
        continue
    
    # Format time as YYYY-MM (similar to the example format)
    if isinstance(time_value, pd.Timestamp):
        time_str = time_value.strftime('%Y-%m')
    else:
        continue
    
    # Extract data for each appliance type
    appliances_data = [
        {
            'appliance': 'Household Refrigerator',
            'current_sales': row.iloc[1],      # Current Period Sales
            'accumulated_sales': row.iloc[2],  # Accumulated Sales
            'yoy_growth': row.iloc[3],         # Year-on-Year Growth
            'accumulated_growth': row.iloc[4]  # Accumulated Growth
        },
        {
            'appliance': 'Air Conditioner', 
            'current_sales': row.iloc[5],      # Current Period Sales
            'accumulated_sales': row.iloc[6],  # Accumulated Sales
            'yoy_growth': row.iloc[7],         # Year-on-Year Growth
            'accumulated_growth': row.iloc[8]  # Accumulated Growth
        },
        {
            'appliance': 'Household Washing Machines',
            'current_sales': row.iloc[9],      # Current Period Sales
            'accumulated_sales': row.iloc[10], # Accumulated Sales
            'yoy_growth': row.iloc[11],        # Year-on-Year Growth
            'accumulated_growth': row.iloc[12] # Accumulated Growth
        }
    ]
    
    # Add each appliance as a separate row
    for appliance_data in appliances_data:
        # Only add rows that have at least some data (not all NaN)
        if not all(pd.isna(val) for val in [appliance_data['current_sales'], 
                                           appliance_data['accumulated_sales'],
                                           appliance_data['yoy_growth'],
                                           appliance_data['accumulated_growth']]):
            transformed_data.append({
                'Time': time_str,
                'Appliance types': appliance_data['appliance'],
                'Current Period Sales(Ten Thousand Units)': appliance_data['current_sales'],
                'Accumulated Sales (Ten Thousand Units)': appliance_data['accumulated_sales'],
                'Year-on-Year Growth (%)': appliance_data['yoy_growth'],
                'Accumulated Growth (%)': appliance_data['accumulated_growth']
            })

# Create DataFrame from transformed data
result_df = pd.DataFrame(transformed_data)

print("=== TRANSFORMATION COMPLETE ===")
print(f"Original data shape: {data_rows.shape}")
print(f"Transformed data shape: {result_df.shape}")
print("\n=== FIRST 15 ROWS OF TRANSFORMED DATA ===")
print(result_df.head(15).to_string(index=False))

print("\n=== LAST 10 ROWS OF TRANSFORMED DATA ===")
print(result_df.tail(10).to_string(index=False))
