    '2023Q1': 'https://13f.info/cusip/67066G104/2023/1',
    '2023Q2': 'https://13f.info/cusip/67066G104/2023/2',
    '2023Q3': 'https://13f.info/cusip/67066G104/2023/3',
    '2023Q4': 'https://13f.info/cusip/67066G104/2023/4',
    '2024Q1': 'https://13f.info/cusip/67066G104/2024/1',
    '2024Q2': 'https://13f.info/cusip/67066G104/2024/2',
    '2024Q3': 'https://13f.info/cusip/67066G104/2024/3',
    '2024Q4': 'https://13f.info/cusip/67066G104/2024/4',
}

# Function to fetch and parse top holders for a quarter
def fetch_top_holders(url, quarter):
    print(f"Fetching data for {quarter}...")
    try:
        response = requests.get(url, timeout=30)
        response.raise_for_status()
        
        soup = BeautifulSoup(response.content, 'html.parser')
        
        # Find the table with holders data
        table = soup.find('table')
        if not table:
            print(f"  No table found for {quarter}")
            return None
        
        # Parse table rows
        holders = []
        rows = table.find_all('tr')[1:]  # Skip header row
        
        for i, row in enumerate(rows[:20]):  # Get top 20
            cols = row.find_all('td')
            if len(cols) >= 5:
                holder_data = {
                    'rank': i + 1,
                    'manager': cols[0].get_text(strip=True),
                    'shares': cols[1].get_text(strip=True),
                    'value': cols[2].get_text(strip=True),
                    'pct_outstanding': cols[3].get_text(strip=True) if len(cols) > 3 else '',
                    'change': cols[4].get_text(strip=True) if len(cols) > 4 else ''
                }
                holders.append(holder_data)
        
        print(f"  Found {len(holders)} holders")
        return holders
        
    except Exception as e:
        print(f"  Error fetching {quarter}: {e}")
        return None

# Fetch data for all quarters
all_holders_data = {}
for quarter, url in quarter_urls.items():
    holders = fetch_top_holders(url, quarter)
    if holders:
        all_holders_data[quarter] = holders
    time.sleep(1)  # Be respectful with requests

# Save the data
with open('/workspace/dumps/workspace/nvda_holders_data.json', 'w') as f:
    json.dump(all_holders_data, f, indent=2)

print(f"\nFetched data for {len(all_holders_data)} quarters")
print("Data saved to nvda_holders_data.json")
