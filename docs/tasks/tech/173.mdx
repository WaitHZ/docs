import requests
import json
from datetime import datetime

def get_repo_info(repo_id):
    """Fetch repository information using GitHub API"""
    url = f"https://api.github.com/repositories/{repo_id}"
    
    try:
        response = requests.get(url)
        if response.status_code == 200:
            data = response.json()
            
            # Extract the required information
            repo_info = {
                "repo_name": data.get("name", ""),
                "owner": data.get("owner", {}).get("login", ""),
                "star_count": data.get("stargazers_count", 0),
                "fork_count": data.get("forks_count", 0),
                "creation_time": data.get("created_at", ""),
                "description": data.get("description", ""),
                "language": data.get("language", ""),
                "repo_url": data.get("html_url", "")
            }
            
            return repo_info
        else:
            print(f"Failed to fetch repo {repo_id}: HTTP {response.status_code}")
            return None
            
    except Exception as e:
        print(f"Error fetching repo {repo_id}: {str(e)}")
        return None

# List of milestone repo IDs
milestone_repos = [1, 1000, 1000000, 1000000000]

# Dictionary to store all repo information
github_info = {}

print("Fetching GitHub repository information...")
print("=" * 50)

for repo_id in milestone_repos:
    print(f"Fetching repo ID: {repo_id}")
    repo_info = get_repo_info(repo_id)
    
    if repo_info:
        github_info[str(repo_id)] = repo_info
        print(f"‚úì Successfully fetched: {repo_info['owner']}/{repo_info['repo_name']}")
    else:
        print(f"‚úó Failed to fetch repo ID: {repo_id}")
    
    print()

print(f"Successfully collected information for {len(github_info)} repositories")
print("Repository information:")
for repo_id, info in github_info.items():
    print(f"- Repo {repo_id}: {info['owner']}/{info['repo_name']} ({info['star_count']} stars)")
