from collections import defaultdict

# Read the orders file
with open('/workspace/dumps/workspace/.overlong_tool_outputs/aAt6vSk35YHqAKWpyWB5HW.json', 'r') as f:

orders = json.loads(orders_data['content'])

# Track variation sales by product
# Structure: {product_id: {variation_id: {quantity_sold, image_id, image_src}}}
product_variations = defaultdict(lambda: defaultdict(lambda: {'quantity': 0, 'image_id': None, 'image_src': None}))

for order in orders:
    for item in order.get('line_items', []):
        product_id = item.get('product_id')
        variation_id = item.get('variation_id')
        quantity = item.get('quantity', 0)

            product_variations[product_id][variation_id]['quantity'] += quantity

            if item.get('image'):

# Find best-selling variation for each product
best_selling_variations = {}
for product_id, variations in product_variations.items():
