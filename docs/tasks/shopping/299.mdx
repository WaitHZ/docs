    template = f.read()

lines = template.split('\n')
subject_line = ""
body_start = 0

for i, line in enumerate(lines):
    if line.startswith('## Email Subject'):
        subject_line = lines[i+1].strip()
    if line.startswith('## Email Body'):
        body_start = i + 1
        break

email_body_template = '\n'.join(lines[body_start:]).strip()

        'body': email_body
