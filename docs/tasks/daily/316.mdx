for i, row in enumerate(rows[1:], start=1):  # Skip header row
    if 'values' not in row or len(row['values']) < 6:
        continue
    
    values = row['values']
    
    # Extract date
    if 'formattedValue' in values[0]:
        date = values[0]['formattedValue']
    else:
        continue
    
    # Extract visitor team
    if len(values) > 3 and 'formattedValue' in values[3]:
        visitor = values[3]['formattedValue']
    else:
        continue
    
    # Extract home team
    if len(values) > 5 and 'formattedValue' in values[5]:
        home = values[5]['formattedValue']

    # Add to visitor's schedule (away game)

        'date': date,
        'location': 'Away',
        'opponent': home
    })
    
    # Add to home team's schedule (home game)

        'date': date,
        'location': 'Home',
        'opponent': visitor
    })

    team_schedules[team].sort(key=lambda x: x['date'])

# Calculate back-to-back statistics for each team
b2b_stats = {}

for team, schedule in team_schedules.items():
    ha_count = 0  # Home-Away
    ah_count = 0  # Away-Home
    hh_count = 0  # Home-Home
    aa_count = 0  # Away-Away
    
    # Look for consecutive games
    for i in range(len(schedule) - 1):
        current_game = schedule[i]
        next_game = schedule[i + 1]
        
        # Parse dates
        current_date = datetime.strptime(current_game['date'], '%Y-%m-%d')
        next_date = datetime.strptime(next_game['date'], '%Y-%m-%d')
        
        # Check if games are on consecutive days (back-to-back)
        if (next_date - current_date).days == 1:
            current_loc = current_game['location']
            next_loc = next_game['location']
            
            if current_loc == 'Home' and next_loc == 'Away':
                ha_count += 1
            elif current_loc == 'Away' and next_loc == 'Home':
                ah_count += 1
            elif current_loc == 'Home' and next_loc == 'Home':
                hh_count += 1
            elif current_loc == 'Away' and next_loc == 'Away':
                aa_count += 1
    
    total = ha_count + ah_count + hh_count + aa_count
    
    b2b_stats[team] = {
        'HA': ha_count,
        'AH': ah_count,
        'HH': hh_count,
        'AA': aa_count,
        'Total': total
    }
